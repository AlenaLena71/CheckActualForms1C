Процедура ПроверитьАктуальностьПечатныхФорм(Команда)
    
    // Открываем диалог выбора файла для загрузки Excel-файла печатной формы
    ДиалогВыбораФайла = Новый ДиалогВыбораФайла(РежимДиалогаВыбораФайла.Открытие);
    ДиалогВыбораФайла.Заголовок = "Выберите файл печатной формы";
    ДиалогВыбораФайла.Фильтр = "Файлы Excel (*.xls;*.xlsx)|*.xls;*.xlsx";
    
    Если Не ДиалогВыбораФайла.Выбрать() Тогда
        Возврат; // Пользователь отменил выбор файла
    КонецЕсли;
    
    ПутьКФайлу = ДиалогВыбораФайла.ПолноеИмяФайла;
    
    // Вызываем клиентскую функцию, которая читает Excel и запускает поиск в макетах
    Результат = ПроверитьФайлИНайтиВБазе(ПутьКФайлу);
    
    // Обработка ошибок чтения Excel
    Если Не Результат.Успех Тогда
        ПоказатьПредупреждение(, Результат.ТекстОшибки);
        Возврат;
    КонецЕсли;
    
    // Проверка, что в файле нашли одну из основных фраз (заголовок формы)
    Если Не Результат.ЗаголовокНайден Тогда
        ПоказатьПредупреждение(, "Неправильно оформленная печатная форма (Отсутствует заголовок)");
        Возврат;
    КонецЕсли;
    
    // Проверка, что в той же ячейке присутствует подфраза (утверждение/письмо)
    Если Не Результат.ЕстьПодфразы Тогда
        ПоказатьПредупреждение(, "Неправильно оформленная печатная форма (Заголовок не объединён в одну ячейку)");
        Возврат;
    КонецЕсли;
    
    // Вывод результата поиска в базе макетов
    Если Результат.ЕстьВБазе Тогда
        ПоказатьПредупреждение(, "Данная печатная форма уже есть в базе");
        // Уточняем, в какой обработке и макете найдено совпадение
        Сообщить("Найдено в обработке: " + Результат.ИмяОбработкиНайденной + ", макет: " + Результат.ИмяМакетаНайденного);
    Иначе
        ПоказатьПредупреждение(, "Данной печатной формы нет в базе");
    КонецЕсли;
    
КонецПроцедуры

&НаКлиенте
Функция ПроверитьФайлИНайтиВБазе(ПутьКФайлу)
    
    // Структура для сбора всех результатов проверки
    Результат = Новый Структура;
    Результат.Вставить("Успех", Ложь);
    Результат.Вставить("ТекстОшибки", "");
    Результат.Вставить("ЗаголовокНайден", Ложь);
    Результат.Вставить("ЕстьПодфразы", Ложь);
    Результат.Вставить("ЕстьВБазе", Ложь);
    Результат.Вставить("ИмяОбработкиНайденной", "");
    Результат.Вставить("ИмяМакетаНайденного", "");
    
    ТекстЗаголовка = ""; // Временная переменная для хранения полного текста ячейки с заголовком
    
    Попытка
        // Создаём COM-объект для работы с Excel (требуется установленный Excel на клиенте)
        Excel = Новый COMОбъект("Excel.Application");
        Excel.Visible = Ложь;      // Не показываем окно Excel
        Excel.DisplayAlerts = Ложь; // Отключаем предупреждения Excel
        
        Книга = Excel.Workbooks.Open(ПутьКФайлу);
        Лист = Книга.Worksheets(1); // Берём первый лист книги
        
        // Перечень основных фраз, которые характеризуют заголовок печатной формы
        Фразы = Новый Массив;
        Фразы.Добавить("Типовая межотраслевая форма №");
        Фразы.Добавить("Унифицированная форма №");
        Фразы.Добавить("Приложение №");
        
        // Перечень подфраз, которые подтверждают, что форма утверждена официально
        Подфразы = Новый Массив;
        Подфразы.Добавить("Утверждена постановлением Госкомстата России от");
        Подфразы.Добавить("к письму ФНС России от");
        Подфразы.Добавить("Правительства Российской Федерации от");
        
        // Определяем фактически используемый диапазон (до 20 строк и 300 столбцов)
        ИспользуемыйДиапазон = Лист.UsedRange;
        МаксСтрока = Мин(ИспользуемыйДиапазон.Rows.Count, 20);
        МаксКолонка = Мин(ИспользуемыйДиапазон.Columns.Count, 300);
        
        Найден = Ложь;
        // Обходим ячейки построчно
        Для Стр = 1 По МаксСтрока Цикл
            Для Кол = 1 По МаксКолонка Цикл
                Попытка
                    Значение = Лист.Cells(Стр, Кол).Value;
                    Если Значение <> Неопределено Тогда
                        ТекстЯчейки = СокрЛП(Строка(Значение));
                        // Ищем любую основную фразу
                        Для Каждого Фраза Из Фразы Цикл
                            Если Найти(ВРег(ТекстЯчейки), ВРег(Фраза)) > 0 Тогда
                                Результат.ЗаголовокНайден = Истина;
                                ТекстЗаголовка = ТекстЯчейки;
                                Найден = Истина;
                                Прервать;
                            КонецЕсли;
                        КонецЦикла;
                        
                        Если Найден Тогда
                            // Проверяем наличие подфразы в этой же ячейке
                            Для Каждого Подфраза Из Подфразы Цикл
                                Если Найти(ВРег(ТекстЯчейки), ВРег(Подфраза)) > 0 Тогда
                                    Результат.ЕстьПодфразы = Истина;
                                    Прервать;
                                КонецЕсли;
                            КонецЦикла;
                            Прервать;
                        КонецЕсли;
                    КонецЕсли;
                Исключение
                    Продолжить; // Пропускаем ячейки с ошибками чтения
                КонецПопытки;
            КонецЦикла;
            Если Найден Тогда
                Прервать;
            КонецЕсли;
        КонецЦикла;
        
        Книга.Close(Ложь); // Закрываем книгу без сохранения
        Excel.Quit();      // Закрываем приложение Excel
        Результат.Успех = Истина;
        
    Исключение
        // Обрабатываем ошибку, возникшую при чтении Excel
        ТекстОшибки = ПодробноеПредставлениеОшибки(ИнформацияОбОшибке());
        Попытка
            Если ЗначениеЗаполнено(Книга) Тогда Книга.Close(Ложь); КонецЕсли;
            Если ЗначениеЗаполнено(Excel) Тогда Excel.Quit(); КонецЕсли;
        Исключение
        КонецПопытки;
        Результат.ТекстОшибки = "Ошибка при чтении файла Excel: " + ТекстОшибки;
        Возврат Результат;
    КонецПопытки;
    
    Если Результат.ЗаголовокНайден И Результат.ЕстьПодфразы Тогда
        // Нормализуем найденный текст: удаляем лишние пробелы, переносы строк и т.п.
        ИскомыйТекст = НормализоватьТекстНаКлиенте(ТекстЗаголовка);
        // Передаём нормализованный текст на сервер для поиска в макетах
        РезультатПоиска = ПроверитьНаличиеПолногоТекстаВМакетах(ИскомыйТекст);
        Результат.ЕстьВБазе = РезультатПоиска.ЕстьВБазе;
        Результат.ИмяОбработкиНайденной = РезультатПоиска.ИмяОбработкиНайденной;
        Результат.ИмяМакетаНайденного = РезультатПоиска.ИмяМакетаНайденного;
    КонецЕсли;
    
    Возврат Результат;
    
КонецФункции

&НаКлиенте
// Функция нормализации текста на клиенте (аналог серверной, чтобы не гонять лишние вызовы)
Функция НормализоватьТекстНаКлиенте(ИсходныйТекст)
    
    Если ИсходныйТекст = Неопределено Тогда
        Возврат "";
    КонецЕсли;
    
    // Заменяем все нестандартные пробельные символы на обычный пробел
    Результат = СтрЗаменить(ИсходныйТекст, Символы.ПС, " ");  // Перевод строки
    Результат = СтрЗаменить(Результат, Символы.Таб, " ");    // Табуляция
    Результат = СтрЗаменить(Результат, Символы.НПП, " ");    // Неразрывный пробел
    Результат = СтрЗаменить(Результат, Символы.ВК, " ");     // Возврат каретки
    
    // Убираем множественные пробелы
    Пока Найти(Результат, "  ") > 0 Цикл
        Результат = СтрЗаменить(Результат, "  ", " ");
    КонецЦикла;
    
    Возврат СокрЛП(Результат);
    
КонецФункции

Функция ПроверитьНаличиеПолногоТекстаВМакетах(ИскомыйТекст)
    
    // Включаем привилегированный режим, чтобы получить доступ ко всем макетам
    УстановитьПривилегированныйРежим(Истина);
    
    Сообщить("--- Поиск полного текста в макетах обработок ---");
    Сообщить("Искомый текст: " + ИскомыйТекст);
    
    КоличествоПроверенных = 0;
    // Структура для возврата результатов поиска на сервере
    РезультатПоиска = Новый Структура("ЕстьВБазе, ИмяОбработкиНайденной, ИмяМакетаНайденного", Ложь, "", "");
    
    // Перебираем все обработки, зарегистрированные в метаданных
    Для Каждого ОбработкаМета Из Метаданные.Обработки Цикл
        // Для каждой обработки перебираем все её макеты
        Для Каждого Макет Из ОбработкаМета.Макеты Цикл
            Попытка
                КоличествоПроверенных = КоличествоПроверенных + 1;
                
                // Получаем менеджер обработки по имени, чтобы получить доступ к макетам
                МенеджерОбработки = Обработки[ОбработкаМета.Имя];
                Если МенеджерОбработки = Неопределено Тогда
                    Продолжить;
                КонецЕсли;
                
                МакетОбъект = МенеджерОбработки.ПолучитьМакет(Макет.Имя);
                Если МакетОбъект = Неопределено Тогда
                    Продолжить;
                КонецЕсли;
                
                ДанныеМакета = Неопределено;
                Попытка
                    ДанныеМакета = МакетОбъект.Получить();
                Исключение
                    // Если метод Получить() недоступен, используем сам объект макета
                КонецПопытки;
                
                Если ДанныеМакета = Неопределено Тогда
                    ДанныеМакета = МакетОбъект;
                КонецЕсли;
                
                // Проверяем, содержится ли искомый текст в данных макета
                Если НайтиПолныйТекстВМакете(ДанныеМакета, ИскомыйТекст) Тогда
                    Сообщить("*** Найдено совпадение! ***");
                    РезультатПоиска.ЕстьВБазе = Истина;
                    РезультатПоиска.ИмяОбработкиНайденной = ОбработкаМета.Имя;
                    РезультатПоиска.ИмяМакетаНайденного = Макет.Имя;
                    Возврат РезультатПоиска;
                КонецЕсли;
                
            Исключение
                // Ошибки при проверке макета записываем в журнал, но не останавливаем перебор
                ЗаписьЖурналаРегистрации(
                    "Поиск печатной формы в макетах",
                    УровеньЖурналаРегистрации.Ошибка,
                    ОбработкаМета.Имя,
                    Макет.Имя,
                    ПодробноеПредставлениеОшибки(ИнформацияОбОшибке())
                );
            КонецПопытки;
        КонецЦикла;
    КонецЦикла;
    
    Сообщить("Проверено макетов: " + КоличествоПроверенных);
    Сообщить("--- Поиск завершён, совпадений не найдено ---");
    Возврат РезультатПоиска;
    
КонецФункции

Функция НайтиПолныйТекстВМакете(ДанныеМакета, ИскомыйТекст)
    
    ТипДанных = ТипЗнч(ДанныеМакета);
    
    Если ТипДанных = Тип("ТабличныйДокумент") Тогда
        // Если макет — табличный документ, ищем по всем ячейкам
        Возврат ИскатьТекстВТабличномДокументе(ДанныеМакета, ИскомыйТекст);
        
    ИначеЕсли ТипДанных = Тип("ТекстовыйДокумент") Тогда
        // Текстовый макет: очищаем от HTML, нормализуем и ищем
        ТекстМакета = ДанныеМакета.ПолучитьТекст();
        ТекстБезТегов = УдалитьHTMLТеги(ТекстМакета);
        ТекстНорм = НормализоватьТекст(ТекстБезТегов);
        Возврат Найти(ВРег(ТекстНорм), ВРег(ИскомыйТекст)) > 0;
        
    ИначеЕсли ТипДанных = Тип("ДвоичныеДанные") Тогда
        // Двоичные данные могут быть mxl-таблицей или текстом в разных кодировках
        Попытка
            ТабДок = Новый ТабличныйДокумент;
            ТабДок.Прочитать(ДанныеМакета);
            Возврат ИскатьТекстВТабличномДокументе(ТабДок, ИскомыйТекст);
        Исключение
        КонецПопытки;
        
        // Пробуем как текст UTF-8
        Попытка
            ТекстUTF8 = ПолучитьСтрокуИзДвоичныхДанных(ДанныеМакета, КодировкаТекста.UTF8);
            ТекстБезТегов = УдалитьHTMLТеги(ТекстUTF8);
            ТекстНорм = НормализоватьТекст(ТекстБезТегов);
            Если Найти(ВРег(ТекстНорм), ВРег(ИскомыйТекст)) > 0 Тогда
                Возврат Истина;
            КонецЕсли;
        Исключение
        КонецПопытки;
        
        // Пробуем как текст ANSI
        Попытка
            ТекстANSI = ПолучитьСтрокуИзДвоичныхДанных(ДанныеМакета, КодировкаТекста.ANSI);
            ТекстБезТегов = УдалитьHTMLТеги(ТекстANSI);
            ТекстНорм = НормализоватьТекст(ТекстБезТегов);
            Если Найти(ВРег(ТекстНорм), ВРег(ИскомыйТекст)) > 0 Тогда
                Возврат Истина;
            КонецЕсли;
        Исключение
        КонецПопытки;
    КонецЕсли;
    
    Возврат Ложь;
КонецФункции

Функция ИскатьТекстВТабличномДокументе(ТабДок, ИскомыйТекст)
    
    Для Стр = 1 По ТабДок.ВысотаТаблицы Цикл

                ТекстЯчейки = ТабДок.Область(Стр, Кол).Текст;

                    ТекстНорм = НормализоватьТекст(ТекстЯчейки);
                    Если Найти(ВРег(ТекстНорм), ВРег(ИскомыйТекст)) > 0 Тогда
                        Возврат Истина;
                    КонецЕсли;
                КонецЕсли;
            Исключение
                // Игнорируем ошибки доступа к отдельным ячейкам
            КонецПопытки;
        КонецЦикла;
    КонецЦикла;
    
    Возврат Ложь;
КонецФункции

// Удаляет все HTML-теги и заменяет основные HTML-сущности на соответствующие символы
Функция УдалитьHTMLТеги(Текст)
    
    Результат = Текст;
    МаксИтераций = 1000; // Защита от бесконечного цикла на случай некорректной разметки
    Счетчик = 0;
    Пока Найти(Результат, "<") > 0 И Найти(Результат, ">") > 0 И Счетчик < МаксИтераций Цикл
        Начало = Найти(Результат, "<");
        Конец = Найти(Результат, ">");
        Если Конец > Начало Тогда
            Результат = Лев(Результат, Начало - 1) + Сред(Результат, Конец + 1);
        Иначе
            Прервать;
        КонецЕсли;
        Счетчик = Счетчик + 1;
    КонецЦикла;
    
    // Замена наиболее частых HTML-сущностей
    Результат = СтрЗаменить(Результат, "&nbsp;", " ");
    Результат = СтрЗаменить(Результат, "&quot;", """");
    Результат = СтрЗаменить(Результат, "&amp;", "&");
    Результат = СтрЗаменить(Результат, "&lt;", "<");
    Результат = СтрЗаменить(Результат, "&gt;", ">");
    
    Возврат Результат;
    
КонецФункции

    Если ИсходныйТекст = Неопределено Тогда
        Возврат "";
    КонецЕсли;

    Результат = СтрЗаменить(Результат, Символы.Таб, " ");
    Результат = СтрЗаменить(Результат, Символы.НПП, " ");
    Результат = СтрЗаменить(Результат, Символы.ВК, " ");
    
    Пока Найти(Результат, "  ") > 0 Цикл
        Результат = СтрЗаменить(Результат, "  ", " ");
    КонецЦикла;
    
    Возврат СокрЛП(Результат);
    
КонецФункции
